 │   │   │   ├─ DietDao.java             // 식단 CRUD DAO 인터페이스

 │   │       ├─ ChallengeServiceImpl.java // 챌린지 서비스 구현
 │   │       ├─ CommunityServiceImpl.java // 커뮤니티 서비스 구현
